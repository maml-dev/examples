# 762 - Docker Compose

Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:20-slim"
      ports: [
        "3935:3441"
      ]
      environment: {
        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/invite"
      }
      depends_on: [
        "db"
        "redis"
      ]
    }

    db: {
      image: "mariadb:11"
      volumes: [
        "db_data:/var/lib/postgresql/data"
      ]
      environment: {
        POSTGRES_DB: "fencing"
        POSTGRES_PASSWORD: "TPOPRa69_WOx9rU"
      }
    }

    redis: {
      image: "redis:7-alpine"
      ports: [
        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
