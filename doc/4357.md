# 4357 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "viewer"
      permissions: [
        "export_data"
        "view_analytics"
        "read"
      ]
      max_seats: 23
    }

    {
      name: "moderator"
      permissions: [
        "manage_billing"
        "delete"
        "read"
        "manage_users"
        "export_data"
      ]
      max_seats: 46
    }

    {
      name: "billing"
      permissions: [
        "manage_billing"
        "read"
      ]
      max_seats: 45
    }

    {
      name: "editor"
      permissions: [
        "write"
        "export_data"
        "manage_billing"
        "read"
        "delete"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "view_analytics"
        "read"
        "manage_billing"
        "export_data"
        "write"
      ]
      max_seats: null
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "read"
        "delete"
      ]
      max_seats: null
    }
  ]
}
```
