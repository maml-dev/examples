# 1851 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "viewer"
      permissions: [
        "manage_users"
        "read"
        "export_data"
        "delete"
        "view_analytics"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "view_analytics"
        "read"
        "delete"
        "manage_billing"
      ]
      max_seats: null
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "manage_billing"
        "delete"
        "read"
        "manage_users"
      ]
      max_seats: 14
    }

    {
      name: "viewer"
      permissions: [
        "delete"
        "export_data"
        "read"
        "view_analytics"
        "manage_billing"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "export_data"
        "read"
        "manage_billing"
      ]
      max_seats: 12
    }
  ]
}
```
