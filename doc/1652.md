# 1652 - Cron Jobs

Scheduled task definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  timezone: "America/New_York"
  jobs: [
    {
      name: "backup-db"
      schedule: "0 * * * *"
      command: "node scripts/cleanup.js"
      enabled: false
    }

    {
      name: "backup-db"
      schedule: "30 6 * * 1-5"
      command: "node scripts/cleanup.js"
      enabled: false
    }

    {
      name: "sync-data"
      schedule: "0 * * * *"
      command: "node scripts/cleanup.js"
      enabled: true
    }

    {
      name: "send-reports"
      schedule: "0 2 * * 0"
      command: "./bin/sync --all"
      enabled: true
    }

    {
      name: "cleanup-temp"
      schedule: "0 2 * * 0"
      command: "./bin/sync --all"
      enabled: true
    }

    {
      name: "rotate-logs"
      schedule: "*/15 * * * *"
      command: "./bin/sync --all"
      enabled: true
    }
  ]
}
```
