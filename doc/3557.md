# 3557 - API Routes

REST API route definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  base_url: "/api/v1"
  routes: [
    {
      method: "PATCH"
      path: "/products"
      handler: "createUsers"
      auth: true
    }

    {
      method: "PATCH"
      path: "/posts/:id"
      handler: "getProducts"
      auth: true
    }

    {
      method: "PATCH"
      path: "/posts/:id"
      handler: "updateUsers"
      auth: true
    }

    {
      method: "PUT"
      path: "/products"
      handler: "deleteOrders"
      auth: false
    }

    {
      method: "DELETE"
      path: "/orders/:id"
      handler: "listProducts"
      auth: false
    }

    {
      method: "PUT"
      path: "/users"
      handler: "deleteOrders"
      auth: false
    }

    {
      method: "POST"
      path: "/products"
      handler: "deleteUsers"
      auth: false
    }
  ]
}
```
