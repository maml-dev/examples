# 4262 - Notification Config

Multi-channel notification routing rules. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  channels: {
    email: {
      enabled: true
      provider: "mailgun"
      from: "notifications@finished-morbidity.biz"
    }

    slack: {
      enabled: false
      webhook_url: "https://hooks.slack.com/services/LPatxoo5eNqNSGfcudbB"
      default_channel: "#general"
    }

    sms: {
      enabled: false
      provider: "sns"
    }
  }

  # Routing rules
  rules: [
    {
      event: "deploy_success"
      channels: [
        "slack"
        "email"
      ]
      priority: "low"
    }

    {
      event: "deploy_failure"
      channels: [
        "slack"
        "email"
        "sms"
      ]
      priority: "critical"
    }

    {
      event: "error_spike"
      channels: [
        "slack"
      ]
      priority: "high"
    }
  ]
}
```
