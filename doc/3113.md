Structured logging configuration. This is an example of a [MAML](https://maml.dev) document.

  level: "error"
  format: "logfmt"
  timestamp: true
  caller: true # include file:line

  output: "stderr"
  rotation: true
  max_size_mb: 377
  max_age_days: 87
  compress: true
}
