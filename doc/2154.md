# 2154 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "moderator"
      permissions: [
        "manage_billing"
        "write"
      ]
      max_seats: 21
    }

    {
      name: "billing"
      permissions: [
        "read"
        "manage_billing"
        "write"
        "view_analytics"
      ]
      max_seats: 9
    }

    {
      name: "viewer"
      permissions: [
        "write"
        "manage_billing"
        "delete"
        "manage_users"
        "export_data"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "view_analytics"
        "read"
        "manage_billing"
      ]
      max_seats: 13
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "manage_billing"
      ]
      max_seats: 8
    }

    {
      name: "moderator"
      permissions: [
        "write"
        "manage_users"
        "delete"
        "read"
      ]
      max_seats: null
    }
  ]
}
```
