# 121 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "digitalocean"
  region: "eu-west-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.138.0.0/16"
      enable_dns: true
      tags: {
        Name: "haversack-vpc"
        Environment: "staging"
      }
    }

    # Compute
    instance: {
      ami: "ami-cfdcf48c"
      type: "t3.medium"
      key_name: "knickers-key"
      monitoring: true
      root_volume: {
        size: 20
        type: "gp2"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-all"
      ingress_port: 8080
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
