# 3570 - Docker Compose

Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:22"
      ports: [
        "3322:3689"
      ]
      environment: {
        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/community"
      }

        "db"
        "redis"
      ]
    }

    db: {
      image: "mariadb:11"

        "db_data:/var/lib/postgresql/data"

      image: "redis:7-alpine"

        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
