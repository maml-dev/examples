# 2948 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "viewer"
      permissions: [
        "manage_billing"
        "export_data"
        "delete"
        "read"
      ]
      max_seats: null
    }

    {
      name: "viewer"
      permissions: [
        "read"
        "manage_users"
        "manage_billing"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "read"
        "export_data"
        "view_analytics"
        "manage_billing"
      ]
      max_seats: 33
    }

    {
      name: "admin"
      permissions: [
        "export_data"
        "delete"
        "view_analytics"
        "write"
        "manage_billing"
      ]
      max_seats: 20
    }

    {
      name: "moderator"
      permissions: [
        "delete"
        "manage_billing"
        "export_data"
        "read"
        "manage_users"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "write"
        "view_analytics"
      ]
      max_seats: null
    }
  ]
}
```
