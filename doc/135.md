# 135 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "digitalocean"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.75.0.0/16"
      enable_dns: true
      tags: {
        Name: "sand-vpc"
        Environment: "production"
      }
    }

    # Compute
    instance: {
      ami: "ami-e3d4ee8a"
      type: "t3.medium"
      key_name: "valley-key"
      monitoring: true
      root_volume: {
        size: 50
        type: "gp3"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-web"
      ingress_port: 22
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
