# 3548 - DNS Records

DNS zone record entries. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  zone: "educated-thigh.net"
  records: [
    {
      type: "NS"
      name: "*"
      value: "v=spf1 include:_spf.gleaming-hovercraft.com ~all"
      ttl: 300
    }

    {
      type: "CNAME"
      name: "*"
      value: "disloyal-declaration.info"
      ttl: 3600
    }

    {
      type: "NS"
      name: "*"
      value: "passionate-advancement.info"
      ttl: 86400
    }

    {
      type: "A"
      name: "mail"
      value: "8.133.160.25"
      ttl: 86400
    }

    {
      type: "MX"
      name: "api"
      value: "dreary-gym.biz"
      ttl: 300
    }

    {
      type: "A"
      name: "api"
      value: "v=spf1 include:_spf.complicated-bathhouse.info ~all"
      ttl: 86400
    }

    {
      type: "A"
      name: "staging"
      value: "v=spf1 include:_spf.tame-sport.net ~all"
      ttl: 3600
    }
  ]
}
```
