```maml
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.44.0.0/16"
      enable_dns: true
      tags: {
        Name: "precedent-vpc"
        Environment: "staging"
      }
    }

    # Compute

      ami: "ami-22ced7b7"
      type: "t3.small"
      key_name: "information-key"
      monitoring: true
      root_volume: {
        size: 50
        type: "io1"
        encrypted: true

    # Security
    security_group: {
      name: "allow-all"
      ingress_port: 443
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
