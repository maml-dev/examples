# 2851 - API Routes

REST API route definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  base_url: "/api/v2"
  routes: [
    {
      method: "DELETE"
      path: "/products/:id"
      handler: "deletePosts"
      auth: false
    }

    {
      method: "GET"
      path: "/products/:id"
      handler: "updatePosts"
      auth: true
    }

    {
      method: "PUT"
      path: "/posts"
      handler: "deleteOrders"
      auth: false
    }

    {
      method: "POST"
      path: "/posts/:id"
      handler: "deleteUsers"
      auth: false
    }

    {
      method: "DELETE"
      path: "/orders"
      handler: "deleteOrders"
      auth: true
    }

    {
      method: "PUT"
      path: "/orders"
      handler: "deleteOrders"
      auth: true
    }

    {
      method: "GET"
      path: "/comments"
      handler: "updatePosts"
      auth: true
    }
  ]
}
```
