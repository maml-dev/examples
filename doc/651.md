# 651 - Cron Jobs

Scheduled task definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  timezone: "America/New_York"
  jobs: [
    {
      name: "backup-db"
      schedule: "30 6 * * 1-5"
      command: "./bin/sync --all"
      enabled: false
    }

    {
      name: "rotate-logs"
      schedule: "0 0 * * *"
      command: "./bin/sync --all"
      enabled: true
    }

    {
      name: "rotate-logs"
      schedule: "0 0 * * *"
      command: "python manage.py report"
      enabled: false
    }

    {
      name: "backup-db"
      schedule: "0 * * * *"
      command: "python manage.py report"
      enabled: true
    }

    {
      name: "cleanup-temp"
      schedule: "0 0 * * *"
      command: "./bin/sync --all"
      enabled: false
    }

    {
      name: "backup-db"
      schedule: "0 0 * * *"
      command: "node scripts/cleanup.js"
      enabled: true
    }
  ]
}
```
