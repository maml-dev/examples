# 1725 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "google"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.85.0.0/16"
      enable_dns: true
      tags: {
        Name: "coal-vpc"
        Environment: "production"
      }
    }

    # Compute
    instance: {
      ami: "ami-aabf5ee1"
      type: "t3.medium"
      key_name: "citizen-key"
      monitoring: true
      root_volume: {
        size: 20
        type: "gp2"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-all"
      ingress_port: 443
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
