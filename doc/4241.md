# 4241 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "moderator"
      permissions: [
        "export_data"
        "manage_users"
        "write"
        "view_analytics"
      ]
      max_seats: 36
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "manage_users"
      ]
      max_seats: null
    }

    {
      name: "viewer"
      permissions: [
        "view_analytics"
        "manage_users"
        "write"
        "delete"
        "read"
      ]
      max_seats: null
    }

    {
      name: "viewer"
      permissions: [
        "view_analytics"
        "delete"
        "read"
        "manage_users"
        "manage_billing"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "manage_users"
        "write"
        "manage_billing"
        "view_analytics"
        "export_data"
      ]
      max_seats: null
    }
  ]
}
```
