# 368 - Docker Compose

Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:22"
      ports: [
        "3260:3129"
      ]
      environment: {
        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/bonfire"
      }
      depends_on: [
        "db"
        "redis"
      ]
    }

    db: {
      image: "mariadb:11"
      volumes: [
        "db_data:/var/lib/postgresql/data"
      ]
      environment: {
        POSTGRES_DB: "exasperation"
        POSTGRES_PASSWORD: "47OefVtjUYFaJjc"
      }
    }

    redis: {
      image: "redis:7-alpine"
      ports: [
        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
