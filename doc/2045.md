# 2045 - DNS Records

DNS zone record entries. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  zone: "metallic-euphonium.com"
  records: [
    {
      type: "MX"
      name: "staging"
      value: "v=spf1 include:_spf.stunning-flood.org ~all"
      ttl: 3600
    }

    {
      type: "AAAA"
      name: "*"
      value: "v=spf1 include:_spf.trivial-ribbon.com ~all"
      ttl: 300
    }

    {
      type: "CNAME"
      name: "cdn"
      value: "2.188.250.47"
      ttl: 3600
    }

    {
      type: "CNAME"
      name: "mail"
      value: "71.14.76.94"
      ttl: 3600
    }

    {
      type: "MX"
      name: "*"
      value: "aggressive-knickers.biz"
      ttl: 3600
    }

    {
      type: "MX"
      name: "www"
      value: "60.222.230.117"
      ttl: 300
    }

    {
      type: "MX"
      name: "@"
      value: "v=spf1 include:_spf.measly-safe.name ~all"
      ttl: 300
    }

    {
      type: "CNAME"
      name: "staging"
      value: "slight-jellyfish.net"
      ttl: 3600
    }
  ]
}
```
