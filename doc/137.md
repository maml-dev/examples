# 137 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "google"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.87.0.0/16"
      enable_dns: true
      tags: {
        Name: "custom-vpc"
        Environment: "production"
      }
    }

    # Compute
    instance: {
      ami: "ami-cfbc7a05"
      type: "m5.large"
      key_name: "instance-key"
      monitoring: true
      root_volume: {
        size: 20
        type: "gp2"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-web"
      ingress_port: 22
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
