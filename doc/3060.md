# 3060 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "editor"
      permissions: [
        "write"
        "export_data"
        "view_analytics"
        "manage_billing"
        "read"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "manage_billing"
        "view_analytics"
        "read"
        "write"
      ]
      max_seats: 6
    }

    {
      name: "viewer"
      permissions: [
        "view_analytics"
        "export_data"
        "read"
      ]
      max_seats: 39
    }

    {
      name: "viewer"
      permissions: [
        "manage_users"
        "export_data"
      ]
      max_seats: 15
    }

    {
      name: "viewer"
      permissions: [
        "write"
        "manage_billing"
        "export_data"
      ]
      max_seats: 35
    }

    {
      name: "billing"
      permissions: [
        "manage_users"
        "read"
        "manage_billing"
        "export_data"
        "view_analytics"
      ]
      max_seats: 24
    }
  ]
}
```
