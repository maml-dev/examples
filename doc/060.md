# 060 - DNS Records

DNS zone record entries. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  zone: "ethical-freight.biz"
  records: [
    {
      type: "NS"
      name: "@"
      value: "deafening-amendment.net"
      ttl: 3600
    }

    {
      type: "CNAME"
      name: "*"
      value: "dead-subsidy.com"
      ttl: 300
    }

    {
      type: "CNAME"
      name: "staging"
      value: "trusty-railway.com"
      ttl: 86400
    }

    {
      type: "A"
      name: "staging"
      value: "v=spf1 include:_spf.assured-godparent.net ~all"
      ttl: 86400
    }

    {
      type: "TXT"
      name: "www"
      value: "135.205.194.224"
      ttl: 300
    }

    {
      type: "MX"
      name: "mail"
      value: "v=spf1 include:_spf.weekly-heartache.net ~all"
      ttl: 3600
    }

    {
      type: "AAAA"
      name: "@"
      value: "bustling-amnesty.com"
      ttl: 3600
    }

    {
      type: "AAAA"
      name: "mail"
      value: "62.133.144.207"
      ttl: 300
    }
  ]
}
```
