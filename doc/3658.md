# 3658 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "viewer"
      permissions: [
        "manage_billing"
        "read"
        "delete"
        "view_analytics"
      ]
      max_seats: 34
    }

    {
      name: "billing"
      permissions: [
        "export_data"
        "write"
        "delete"
        "read"
        "view_analytics"
      ]
      max_seats: null
    }

    {
      name: "admin"
      permissions: [
        "write"
        "manage_users"
        "read"
        "delete"
        "manage_billing"
      ]
      max_seats: 22
    }

    {
      name: "admin"
      permissions: [
        "export_data"
        "manage_users"
        "manage_billing"
        "write"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "manage_users"
        "export_data"
        "delete"
        "read"
      ]
      max_seats: 24
    }

    {
      name: "support"
      permissions: [
        "view_analytics"
        "manage_users"
        "manage_billing"
        "export_data"
        "read"
      ]
      max_seats: null
    }
  ]
}
```
