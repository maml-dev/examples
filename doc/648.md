# 648 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "admin"
      permissions: [
        "write"
        "manage_billing"
        "view_analytics"
        "manage_users"
        "delete"
      ]
      max_seats: 29
    }

    {
      name: "moderator"
      permissions: [
        "manage_users"
        "delete"
      ]
      max_seats: null
    }

    {
      name: "admin"
      permissions: [
        "manage_billing"
        "manage_users"
        "delete"
        "read"
        "export_data"
      ]
      max_seats: 3
    }

    {
      name: "viewer"
      permissions: [
        "view_analytics"
        "export_data"
        "read"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "read"
        "write"
      ]
      max_seats: 50
    }
  ]
}
```
