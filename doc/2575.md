# 2575 - Docker Compose

Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:22"
      ports: [
        "3880:3073"
      ]
      environment: {
        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/accelerator"
      }
      depends_on: [
        "db"
        "redis"
      ]
    }

    db: {
      image: "mysql:8"
      volumes: [
        "db_data:/var/lib/postgresql/data"
      ]
      environment: {
        POSTGRES_DB: "elevation"
        POSTGRES_PASSWORD: "9ZpksDMwyUNQivI"
      }
    }

    redis: {
      image: "redis:7-alpine"
      ports: [
        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
