# 414 - Locale Config

```maml
  language: "ja"
  country: "SB"
  timezone: "America/Argentina/Salta"
  currency: "AZN"

  date_format: "DD/MM/YYYY"
  time_format: "24h"
  first_day_of_week: "sunday"
  decimal_separator: ","
}
```
