# 4166 - Notification Config

Multi-channel notification routing rules. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  channels: {
    email: {
      enabled: true
      provider: "mailgun"
      from: "notifications@weird-adrenalin.org"
    }

    slack: {
      enabled: true
      webhook_url: "https://hooks.slack.com/services/CzCxxCxGc1BJk2I4yPzj"
      default_channel: "#alerts"
    }

    sms: {
      enabled: false
      provider: "sns"
    }
  }

  # Routing rules
  rules: [
    {
      event: "deploy_success"
      channels: [
        "slack"
        "email"
      ]
      priority: "low"
    }

    {
      event: "deploy_failure"
      channels: [
        "slack"
        "email"
        "sms"
      ]
      priority: "critical"
    }

    {
      event: "error_spike"
      channels: [
        "slack"
      ]
      priority: "high"
    }
  ]
}
```
