# 133 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "google"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.233.0.0/16"
      enable_dns: true
      tags: {
        Name: "dream-vpc"
        Environment: "production"
      }
    }

    # Compute
    instance: {
      ami: "ami-dd96bdbb"
      type: "t3.micro"
      key_name: "spirit-key"
      monitoring: true
      root_volume: {
        size: 50
        type: "gp2"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-ssh"
      ingress_port: 443
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
