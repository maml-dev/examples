# 2141 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "admin"
      permissions: [
        "read"
        "export_data"
      ]
      max_seats: null
    }

    {
      name: "admin"
      permissions: [
        "manage_users"
        "manage_billing"
        "view_analytics"
        "read"
        "export_data"
      ]
      max_seats: 37
    }

    {
      name: "admin"
      permissions: [
        "delete"
        "view_analytics"
        "export_data"
        "write"
        "read"
      ]
      max_seats: 29
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "view_analytics"
      ]
      max_seats: 9
    }

    {
      name: "editor"
      permissions: [
        "export_data"
        "manage_users"
        "delete"
      ]
      max_seats: null
    }

    {
      name: "editor"
      permissions: [
        "manage_users"
        "read"
        "export_data"
        "manage_billing"
      ]
      max_seats: 47
    }
  ]
}
```
