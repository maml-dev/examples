# 642 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "editor"
      permissions: [
        "export_data"
        "manage_users"
        "read"
        "write"
        "view_analytics"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "view_analytics"
        "manage_users"
        "export_data"
      ]
      max_seats: null
    }

    {
      name: "admin"
      permissions: [
        "manage_billing"
        "write"
        "export_data"
        "delete"
        "read"
      ]
      max_seats: null
    }

    {
      name: "viewer"
      permissions: [
        "export_data"
        "manage_billing"
        "view_analytics"
        "write"
        "delete"
      ]
      max_seats: 43
    }

    {
      name: "admin"
      permissions: [
        "delete"
        "write"
      ]
      max_seats: 1
    }

    {
      name: "viewer"
      permissions: [
        "manage_billing"
        "manage_users"
        "export_data"
        "delete"
      ]
      max_seats: null
    }
  ]
}
```
