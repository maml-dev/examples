```maml
    algorithm: "HS256"
    expiry: "1h"
    refresh_expiry: "30d"

  oauth: {
    client_id: "xWq9TBgbNbde2aDTC1CW1UuV"

  session: {
    store: "database"
    cookie_name: "sid"

  }
}
```
