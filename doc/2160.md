# 2160 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "viewer"
      permissions: [
        "manage_billing"
        "manage_users"
        "view_analytics"
        "write"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "export_data"
        "manage_billing"
        "read"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "manage_users"
        "manage_billing"
        "read"
        "view_analytics"
        "write"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "view_analytics"
        "export_data"
        "manage_billing"
        "delete"
        "read"
      ]
      max_seats: 28
    }

    {
      name: "admin"
      permissions: [
        "read"
        "manage_billing"
      ]
      max_seats: null
    }
  ]
}
```
