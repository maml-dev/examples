Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:20-slim"
      ports: [
        "3245:3301"
      ]

        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/seagull"

        "db"
        "redis"
      ]
    }

    db: {
      image: "mysql:8"

        "db_data:/var/lib/postgresql/data"

      image: "redis:7-alpine"

        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
