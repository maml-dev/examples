# 1953 - Cron Jobs

Scheduled task definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  timezone: "America/New_York"
  jobs: [
    {
      name: "send-reports"
      schedule: "0 0 * * *"
      command: "pg_dump $DATABASE_URL"
      enabled: true
    }

    {
      name: "sync-data"
      schedule: "0 2 * * 0"
      command: "pg_dump $DATABASE_URL"
      enabled: true
    }

    {
      name: "send-reports"
      schedule: "*/15 * * * *"
      command: "pg_dump $DATABASE_URL"
      enabled: false
    }

    {
      name: "sync-data"
      schedule: "0 * * * *"
      command: "./bin/sync --all"
      enabled: true
    }

    {
      name: "backup-db"
      schedule: "30 6 * * 1-5"
      command: "python manage.py report"
      enabled: true
    }

    {
      name: "cleanup-temp"
      schedule: "0 * * * *"
      command: "pg_dump $DATABASE_URL"
      enabled: true
    }
  ]
}
```
