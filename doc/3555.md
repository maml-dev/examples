# 3555 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "editor"
      permissions: [
        "delete"
        "manage_billing"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "view_analytics"
        "delete"
        "read"
        "manage_users"
      ]
      max_seats: 35
    }

    {
      name: "viewer"
      permissions: [
        "manage_billing"
        "export_data"
      ]
      max_seats: 44
    }

    {
      name: "admin"
      permissions: [
        "view_analytics"
        "export_data"
        "read"
      ]
      max_seats: 50
    }

    {
      name: "moderator"
      permissions: [
        "write"
        "manage_users"
        "view_analytics"
        "manage_billing"
      ]
      max_seats: 20
    }
  ]
}
```
