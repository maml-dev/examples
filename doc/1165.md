# 1165 - Docker Compose

Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:18-alpine"
      ports: [
        "3128:3639"
      ]
      environment: {
        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/legging"
      }
      depends_on: [
        "db"
        "redis"
      ]
    }

    db: {
      image: "mysql:8"
      volumes: [
        "db_data:/var/lib/postgresql/data"
      ]
      environment: {
        POSTGRES_DB: "tuxedo"
        POSTGRES_PASSWORD: "rd6IFwfZJkSrhh3"
      }
    }

    redis: {
      image: "redis:7-alpine"
      ports: [
        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
