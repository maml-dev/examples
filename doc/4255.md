# 4255 - DNS Records

DNS zone record entries. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  zone: "winding-mainstream.org"
  records: [
    {
      type: "NS"
      name: "api"
      value: "strong-harp.net"
      ttl: 86400
    }

    {
      type: "TXT"
      name: "*"
      value: "old-fashioned-aircraft.com"
      ttl: 86400
    }

    {
      type: "CNAME"
      name: "api"
      value: "v=spf1 include:_spf.zesty-sermon.name ~all"
      ttl: 300
    }

    {
      type: "TXT"
      name: "mail"
      value: "v=spf1 include:_spf.definitive-exploration.biz ~all"
      ttl: 3600
    }

    {
      type: "AAAA"
      name: "*"
      value: "v=spf1 include:_spf.next-galoshes.net ~all"
      ttl: 86400
    }

    {
      type: "A"
      name: "*"
      value: "v=spf1 include:_spf.jumbo-produce.net ~all"
      ttl: 3600
    }

    {
      type: "NS"
      name: "mail"
      value: "125.126.228.26"
      ttl: 300
    }
  ]
}
```
