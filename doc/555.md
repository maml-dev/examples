# 555 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "admin"
      permissions: [
        "write"
        "manage_billing"
        "view_analytics"
        "manage_users"
      ]
      max_seats: null
    }

    {
      name: "moderator"
      permissions: [
        "view_analytics"
        "read"
        "export_data"
        "delete"
      ]
      max_seats: null
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "view_analytics"
        "manage_billing"
        "write"
        "manage_users"
      ]
      max_seats: 44
    }

    {
      name: "moderator"
      permissions: [
        "read"
        "delete"
        "write"
        "view_analytics"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "view_analytics"
        "manage_users"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "delete"
        "manage_users"
        "write"
      ]
      max_seats: 42
    }
  ]
}
```
