# 3152 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "billing"
      permissions: [
        "write"
        "read"
        "view_analytics"
        "manage_users"
        "export_data"
      ]
      max_seats: 34
    }

    {
      name: "billing"
      permissions: [
        "read"
        "manage_users"
        "export_data"
        "manage_billing"
        "write"
      ]
      max_seats: 49
    }

    {
      name: "admin"
      permissions: [
        "manage_users"
        "write"
      ]
      max_seats: 25
    }

    {
      name: "viewer"
      permissions: [
        "export_data"
        "view_analytics"
      ]
      max_seats: 42
    }

    {
      name: "admin"
      permissions: [
        "view_analytics"
        "delete"
        "manage_users"
      ]
      max_seats: 3
    }

    {
      name: "moderator"
      permissions: [
        "manage_billing"
        "read"
        "manage_users"
        "view_analytics"
      ]
      max_seats: 46
    }
  ]
}
```
