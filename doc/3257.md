# 3257 - DNS Records

DNS zone record entries. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  zone: "delectable-smog.name"
  records: [
    {
      type: "AAAA"
      name: "staging"
      value: "v=spf1 include:_spf.hefty-design.info ~all"
      ttl: 86400
    }

    {
      type: "A"
      name: "api"
      value: "224.16.52.203"
      ttl: 3600
    }

    {
      type: "AAAA"
      name: "staging"
      value: "shimmering-solvency.biz"
      ttl: 86400
    }

    {
      type: "NS"
      name: "*"
      value: "v=spf1 include:_spf.necessary-conservation.com ~all"
      ttl: 3600
    }

    {
      type: "MX"
      name: "www"
      value: "95.50.48.110"
      ttl: 86400
    }

    {
      type: "AAAA"
      name: "*"
      value: "frilly-completion.net"
      ttl: 300
    }

    {
      type: "MX"
      name: "cdn"
      value: "v=spf1 include:_spf.pricey-in-joke.org ~all"
      ttl: 300
    }

    {
      type: "AAAA"
      name: "api"
      value: "salty-turret.net"
      ttl: 3600
    }
  ]
}
```
