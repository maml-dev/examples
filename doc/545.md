# 545 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "editor"
      permissions: [
        "delete"
        "read"
      ]
      max_seats: null
    }

    {
      name: "billing"
      permissions: [
        "export_data"
        "read"
      ]
      max_seats: null
    }

    {
      name: "editor"
      permissions: [
        "export_data"
        "manage_billing"
      ]
      max_seats: 19
    }

    {
      name: "admin"
      permissions: [
        "export_data"
        "write"
        "read"
        "manage_users"
        "delete"
      ]
      max_seats: 47
    }

    {
      name: "support"
      permissions: [
        "write"
        "manage_users"
      ]
      max_seats: null
    }

    {
      name: "viewer"
      permissions: [
        "export_data"
        "write"
        "delete"
        "read"
      ]
      max_seats: null
    }
  ]
}
```
