# 4243 - DNS Records

DNS zone record entries. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  zone: "wordy-backburn.name"
  records: [
    {
      type: "NS"
      name: "api"
      value: "84.238.165.159"
      ttl: 300
    }

    {
      type: "A"
      name: "mail"
      value: "v=spf1 include:_spf.silver-agreement.com ~all"
      ttl: 300
    }

    {
      type: "A"
      name: "staging"
      value: "v=spf1 include:_spf.metallic-ceramic.info ~all"
      ttl: 3600
    }

    {
      type: "A"
      name: "@"
      value: "far-flung-editor.biz"
      ttl: 86400
    }

    {
      type: "NS"
      name: "cdn"
      value: "v=spf1 include:_spf.chubby-pants.info ~all"
      ttl: 3600
    }

    {
      type: "MX"
      name: "mail"
      value: "back-procurement.name"
      ttl: 300
    }

    {
      type: "AAAA"
      name: "api"
      value: "fat-toaster.info"
      ttl: 300
    }

    {
      type: "NS"
      name: "*"
      value: "timely-napkin.info"
      ttl: 3600
    }
  ]
}
```
