# 3453 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "editor"
      permissions: [
        "view_analytics"
        "delete"
        "read"
      ]
      max_seats: 5
    }

    {
      name: "editor"
      permissions: [
        "delete"
        "manage_users"
        "view_analytics"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "view_analytics"
        "delete"
        "manage_users"
        "read"
      ]
      max_seats: 17
    }

    {
      name: "support"
      permissions: [
        "delete"
        "manage_users"
        "read"
      ]
      max_seats: 17
    }

    {
      name: "viewer"
      permissions: [
        "view_analytics"
        "manage_users"
        "read"
        "delete"
      ]
      max_seats: 3
    }

    {
      name: "editor"
      permissions: [
        "manage_billing"
        "export_data"
        "delete"
        "manage_users"
      ]
      max_seats: null
    }
  ]
}
```
