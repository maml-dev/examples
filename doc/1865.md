# 1865 - Docker Compose

Multi-container application definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  version: "3.8"
  services: {
    app: {
      image: "node:22"
      ports: [
        "3008:3550"
      ]
      environment: {
        NODE_ENV: "production"
        DATABASE_URL: "postgres://db:5432/sticker"
      }
      depends_on: [
        "db"
        "redis"
      ]
    }

    db: {
      image: "postgres:16"
      volumes: [
        "db_data:/var/lib/postgresql/data"
      ]
      environment: {
        POSTGRES_DB: "story"
        POSTGRES_PASSWORD: "0GxwWtYhHEmWigu"
      }
    }

    redis: {
      image: "redis:7-alpine"
      ports: [
        "6379:6379"
      ]
    }
  }

  volumes: {
    db_data: {}
  }
}
```
