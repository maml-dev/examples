# 333 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "digitalocean"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.70.0.0/16"
      enable_dns: true
      tags: {
        Name: "breastplate-vpc"
        Environment: "production"
      }
    }

    # Compute
    instance: {
      ami: "ami-07bfd5eb"
      type: "t3.small"
      key_name: "peony-key"
      monitoring: true
      root_volume: {
        size: 100
        type: "gp3"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-web"
      ingress_port: 80
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
