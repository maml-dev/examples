# 3124 - Auth Config

Authentication and authorization settings. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "oauth2"
  # Token settings
  jwt: {
    secret_key: "Lxuospv1nXO2jxxqhEHivb3jnDqBOnvn"
    algorithm: "HS256"
    expiry: "24h"
    refresh_expiry: "30d"
  }

  # OAuth provider
  oauth: {
    client_id: "3L7iDFQvYpw4Fi0WaMH5jhaU"
    authorize_url: "https://auth.fruitful-transparency.org/authorize"
    token_url: "https://auth.coarse-travel.com/token"
    scopes: [
      "openid"
      "profile"
      "email"
    ]
  }

  session: {
    store: "database"
    cookie_name: "sid"
    secure: true
    http_only: true
  }
}
```
