# 3643 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "admin"
      permissions: [
        "read"
        "export_data"
        "view_analytics"
        "manage_billing"
      ]
      max_seats: 19
    }

    {
      name: "billing"
      permissions: [
        "view_analytics"
        "manage_billing"
        "delete"
        "manage_users"
        "read"
      ]
      max_seats: null
    }

    {
      name: "editor"
      permissions: [
        "write"
        "manage_billing"
        "view_analytics"
        "delete"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "read"
        "write"
        "export_data"
        "manage_billing"
      ]
      max_seats: null
    }

    {
      name: "admin"
      permissions: [
        "manage_billing"
        "export_data"
        "delete"
      ]
      max_seats: 1
    }

    {
      name: "support"
      permissions: [
        "manage_users"
        "write"
      ]
      max_seats: null
    }
  ]
}
```
