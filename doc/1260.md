# 1260 - API Routes

REST API route definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  base_url: "/api/v3"
  routes: [
    {
      method: "POST"
      path: "/products"
      handler: "getUsers"
      auth: false
    }

    {
      method: "POST"
      path: "/users/:id"
      handler: "updateProducts"
      auth: false
    }

    {
      method: "PATCH"
      path: "/comments"
      handler: "getUsers"
      auth: false
    }

    {
      method: "PATCH"
      path: "/comments"
      handler: "createUsers"
      auth: true
    }

    {
      method: "GET"
      path: "/orders/:id"
      handler: "getOrders"
      auth: false
    }

    {
      method: "PATCH"
      path: "/users"
      handler: "createUsers"
      auth: true
    }

    {
      method: "GET"
      path: "/products"
      handler: "deletePosts"
      auth: true
    }
  ]
}
```
