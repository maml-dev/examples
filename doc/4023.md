# 4023 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "digitalocean"
  region: "eu-west-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.235.0.0/16"
      enable_dns: true
      tags: {
        Name: "monocle-vpc"
        Environment: "staging"
      }
    }

    # Compute
    instance: {
      ami: "ami-20bb4cb3"
      type: "m5.large"
      key_name: "pension-key"
      monitoring: true
      root_volume: {
        size: 20
        type: "io1"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-web"
      ingress_port: 80
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
