# 4233 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "digitalocean"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.84.0.0/16"
      enable_dns: true
      tags: {
        Name: "dredger-vpc"
        Environment: "staging"
      }
    }

    # Compute
    instance: {
      ami: "ami-2abc9ab0"
      type: "m5.large"
      key_name: "pliers-key"
      monitoring: true
      root_volume: {
        size: 20
        type: "gp2"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-ssh"
      ingress_port: 22
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
