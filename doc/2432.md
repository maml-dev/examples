# 2432 - Terraform Resource

Infrastructure as code resource definition. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "digitalocean"
  region: "us-east-1"
  resources: {
    # Network
    vpc: {
      cidr_block: "10.160.0.0/16"
      enable_dns: true
      tags: {
        Name: "privilege-vpc"
        Environment: "production"
      }
    }

    # Compute
    instance: {
      ami: "ami-bf32dec0"
      type: "t3.micro"
      key_name: "import-key"
      monitoring: true
      root_volume: {
        size: 50
        type: "io1"
        encrypted: true
      }
    }

    # Security
    security_group: {
      name: "allow-ssh"
      ingress_port: 8080
      ingress_cidr: "0.0.0.0/0"
    }
  }
}
```
