# 955 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "viewer"
      permissions: [
        "write"
        "export_data"
        "manage_billing"
        "delete"
        "read"
      ]
      max_seats: 33
    }

    {
      name: "moderator"
      permissions: [
        "export_data"
        "delete"
      ]
      max_seats: 15
    }

    {
      name: "billing"
      permissions: [
        "export_data"
        "manage_billing"
        "delete"
        "view_analytics"
        "manage_users"
      ]
      max_seats: 25
    }

    {
      name: "editor"
      permissions: [
        "write"
        "manage_billing"
        "manage_users"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "export_data"
        "delete"
        "view_analytics"
      ]
      max_seats: null
    }
  ]
}
```
