# 129 - Auth Config

Authentication and authorization settings. This is an example of a [MAML](https://maml.dev) document.

```maml
  # Token settings
  jwt: {
    secret_key: "QJOL7wmMQj8st0qfVhykgGjocostT90s"
    algorithm: "ES256"
    expiry: "1h"
    refresh_expiry: "90d"
  }

  # OAuth provider
  oauth: {
    client_id: "VsfpjGUAicF4BCyJ0vl4cdyn"
    authorize_url: "https://auth.cavernous-sonata.net/authorize"
    token_url: "https://auth.woeful-other.com/token"
    scopes: [
      "openid"
      "profile"
      "email"
    ]
  }

  session: {
    store: "redis"
    cookie_name: "sid"
    secure: true
    http_only: true
  }
}
```
