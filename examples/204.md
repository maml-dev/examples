DNS zone configuration. This is an example of a [MAML](https://maml.dev) document.

{
  zone: "critical-lace.biz"
  ttl: 3600 # seconds

    "ns1.boring-exterior.com"
    "ns2.physical-fireplace.biz"

  soa_email: "admin@grumpy-sock.net"

  retry: 900

}
