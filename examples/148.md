# 148 - API Routes

REST API route definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  base_url: "/api/v2"
  routes: [
    {
      method: "DELETE"
      path: "/products"
      handler: "getPosts"
      auth: false
    }

    {
      method: "POST"
      path: "/comments"
      handler: "deleteProducts"
      auth: true
    }

    {
      method: "DELETE"
      path: "/comments/:id"
      handler: "getProducts"
      auth: false
    }

    {
      method: "PATCH"
      path: "/products/:id"
      handler: "deleteUsers"
      auth: false
    }

    {
      method: "DELETE"
      path: "/posts"
      handler: "updateOrders"
      auth: false
    }

    {
      method: "PATCH"
      path: "/orders"
      handler: "createPosts"
      auth: false
    }

    {
      method: "PUT"
      path: "/comments"
      handler: "listOrders"
      auth: false
    }
  ]
}
```
