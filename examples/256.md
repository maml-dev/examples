Typography font family definitions. This is an example of a [MAML](https://maml.dev) document.

{

    {
      family: "Geist"
      category: "display"
      weights: [
        200
        500
        700
        900

      variable: true
    }

      family: "Source Sans Pro"

    }

      family: "Geist"

        200

    }
  ]
}
