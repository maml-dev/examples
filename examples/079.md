# 079 - Notification Config

Multi-channel notification routing rules. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  channels: {
    email: {
      enabled: true
      provider: "postmark"
      from: "notifications@timely-diagram.org"
    }

    slack: {
      enabled: false
      webhook_url: "https://hooks.slack.com/services/Xav6v2yBP0J208dseCkk"
      default_channel: "#deployments"
    }

    sms: {
      enabled: false
      provider: "vonage"
    }
  }

  # Routing rules
  rules: [
    {
      event: "deploy_success"
      channels: [
        "slack"
        "email"
      ]
      priority: "low"
    }

    {
      event: "deploy_failure"
      channels: [
        "slack"
        "email"
        "sms"
      ]
      priority: "critical"
    }

    {
      event: "error_spike"
      channels: [
        "slack"
      ]
      priority: "high"
    }
  ]
}
```
