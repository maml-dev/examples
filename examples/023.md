# 023 - Auth Config

Authentication and authorization settings. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  provider: "oauth2"
  # Token settings
  jwt: {
    secret_key: "LNOr7lszHZbFWJgZtTcG5RB28zqHilYj"
    algorithm: "ES256"
    expiry: "1h"
    refresh_expiry: "30d"
  }

  # OAuth provider
  oauth: {
    client_id: "osMlDfCFAfjIImVq1yFyZgUg"
    authorize_url: "https://auth.lone-pomelo.name/authorize"
    token_url: "https://auth.brilliant-gallery.biz/token"
    scopes: [
      "openid"
      "profile"
      "email"
    ]
  }

  session: {
    store: "redis"
    cookie_name: "sid"
    secure: true
    http_only: true
  }
}
```
