# 243 - Roles & Permissions

Role-based access control definitions. This is an example of a [MAML](https://maml.dev) document.

```maml
{
  roles: [
    {
      name: "support"
      permissions: [
        "manage_users"
        "view_analytics"
        "write"
        "delete"
        "export_data"
      ]
      max_seats: 18
    }

    {
      name: "viewer"
      permissions: [
        "delete"
        "view_analytics"
        "read"
        "export_data"
        "write"
      ]
      max_seats: null
    }

    {
      name: "support"
      permissions: [
        "write"
        "read"
      ]
      max_seats: 34
    }

    {
      name: "moderator"
      permissions: [
        "manage_billing"
        "export_data"
        "view_analytics"
        "read"
        "write"
      ]
      max_seats: null
    }

    {
      name: "admin"
      permissions: [
        "write"
        "export_data"
        "manage_users"
        "manage_billing"
      ]
      max_seats: null
    }
  ]
}
```
